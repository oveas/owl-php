00778     public function moveNode ($field, $value, array $newParent, $position = -1)
00779     {
00780         if (!array_key_exists('field', $newParent) || !array_key_exists('value', $newParent)) {
00781             $this->setStatus(HDATA_IVNODESPEC);
00782             return (false);
00783         }
00784         $table = $this->owl_database->tablename($this->owl_tablename);
00785         $node = $this->getNode($field, $value);
00786         $this->owl_database->startTransaction('insertNode');
00787         $this->owl_database->lockTable($table, DBDRIVER_LOCKTYPE_WRITE);
00788         $_stat = true;
00789 
00790         if ($_stat !== false) {
00791             $_stat = $this->writeQuery("UPDATE $table "
00792                                     . "SET ". $this->owl_database->getDriver()->dbQuote($this->left) . " = 0 "
00793                                     . ",   ". $this->owl_database->getDriver()->dbQuote($this->right) . " = 0 "
00794                                     . "WHERE ". $this->owl_database->getDriver()->dbQuote($this->left) . " = " . $node[$this->left] . ' '
00795                                     , __LINE__);
00796         }
00797         if ($_stat !== false) {
00798             // Set all left and right values in the tree being moved to their negative values
00799             $_stat = $this->writeQuery("UPDATE $table "
00800                                     . "SET ". $this->owl_database->getDriver()->dbQuote($this->left) . " = 0 - ". $this->owl_database->getDriver()->dbQuote($this->left) . " "
00801                                     . ",   ". $this->owl_database->getDriver()->dbQuote($this->right) . " = 0 -". $this->owl_database->getDriver()->dbQuote($this->right) . " "
00802                                     . "WHERE ". $this->owl_database->getDriver()->dbQuote($this->left) . " BETWEEN " . $node[$this->left] . ' AND ' . $node[$this->right] . ' '
00803                                     , __LINE__);
00804         }
00805 
00806         // Get the width of the tree being moved
00807         $_width = ($node[$this->right] - $node[$this->left] + 1);
00808 
00809         if ($_stat !== false) {
00810             // Now the original tree is parked, shrink the total tree by decreasing everything
00811             // right from the original object (in 2 steps to make sure the direct parents right- only
00812             // is decreased!)
00813             $_stat = $this->writeQuery("UPDATE $table "
00814                                     . "SET ". $this->owl_database->getDriver()->dbQuote($this->right) . " = ". $this->owl_database->getDriver()->dbQuote($this->right) . " - $_width "
00815                                     . "WHERE ". $this->owl_database->getDriver()->dbQuote($this->right) . " > " . $node[$this->right] . ' '
00816                                     , __LINE__);
00817         }
00818         if ($_stat !== false) {
00819             $_stat = $this->writeQuery("UPDATE $table "
00820                                     . "SET ". $this->owl_database->getDriver()->dbQuote($this->left) . " = ". $this->owl_database->getDriver()->dbQuote($this->left) . " - $_width "
00821                                     . "WHERE ". $this->owl_database->getDriver()->dbQuote($this->left) . " > " . $node[$this->right] . ' '
00822                                     , __LINE__);
00823         }
00824 
00825         // Now set the new left value. This must be retrieved *after* the shrink above to
00826         // make sure we have the new leftvalue.
00827         // Then calculate how far the tree is actually moved
00828 
00829         // TODO Since _getNewLeft() reads from the table using aliases, we can't access the table
00830         // using the active lock (at least in MySQL), so we temporarily release the lock now
00831         $this->owl_database->unlockTable($table);
00832         $_newLeft = $this->_getNewLeft($newParent, $position);
00833         $this->owl_database->lockTable($table, DBDRIVER_LOCKTYPE_WRITE);
00834         $_newRight = $_newLeft + $_width - 1;
00835         $_move = $node[$this->left] - $_newLeft;
00836 
00837         if ($_stat !== false) {
00838             // Make room for the tree at the new position
00839             $_stat = $this->writeQuery("UPDATE $table "
00840                                     . "SET ". $this->owl_database->getDriver()->dbQuote($this->right) . " = ". $this->owl_database->getDriver()->dbQuote($this->right) . " + $_width "
00841                                     . "WHERE ". $this->owl_database->getDriver()->dbQuote($this->right) . " >= $_newLeft "
00842                                     , __LINE__);
00843         }
00844         if ($_stat !== false) {
00845             $_stat = $this->writeQuery("UPDATE $table "
00846                                     . "SET ". $this->owl_database->getDriver()->dbQuote($this->left) . " = ". $this->owl_database->getDriver()->dbQuote($this->left) . " + $_width "
00847                                     . "WHERE ". $this->owl_database->getDriver()->dbQuote($this->left) . " >= $_newLeft "
00848                                     , __LINE__);
00849         }
00850 
00851         if ($_stat !== false) {
00852             // Ok, now put the root node of this tree back in place
00853             $_stat = $this->writeQuery("UPDATE $table "
00854                                     . "SET ". $this->owl_database->getDriver()->dbQuote($this->left) . " = $_newLeft "
